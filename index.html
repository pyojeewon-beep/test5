import tkinter as tk from tkinter import messagebox import random # 음악 데이터 music_db = { "힙합": [ "Sik-K - party", "ASH ISLAND - Melody", "Kid Milli - WHY DO FUCKBOIS HANG OUT ON THE NET" ], "발라드": [ "아이유 - 밤편지", "성시경 - 희재", "폴킴 - 모든 날, 모든 순간" ], "록": [ "YB - 나는 나비", "NELL - 기억을 걷는 시간", "국카스텐 - 한잔의 추억" ], "팝": [ "The Weeknd - Blinding Lights", "Dua Lipa - Levitating", "Ed Sheeran - Shape of You" ] } # 추천 함수 def recommend_music(): genre = genre_var.get() if genre in music_db: song = random.choice(music_db[genre]) result_label.config( text=f"🎧 추천 음악\n\n{song}", fg="#00ffcc" ) else: messagebox.showwarning("오류", "장르를 선택하세요!") # 메인 창 root = tk.Tk() root.title("🎵 음악 추천 프로그램") root.geometry("500x400") root.configure(bg="#121212") # 제목 title_label = tk.Label( root, text="🎵 MUSIC RECOMMENDER", font=("Helvetica", 24, "bold"), bg="#121212", fg="#ffffff" ) title_label.pack(pady=30) # 설명 desc_label = tk.Label( root, text="좋아하는 장르를 선택하세요", font=("Helvetica", 12), bg="#121212", fg="#bbbbbb" ) desc_label.pack() # 장르 선택 genre_var = tk.StringVar(value="힙합") genre_menu = tk.OptionMenu(root, genre_var, *music_db.keys()) genre_menu.config( font=("Helvetica", 12), bg="#1f1f1f", fg="white", activebackground="#333333", activeforeground="white", width=15 ) genre_menu["menu"].config( bg="#1f1f1f", fg="white" ) genre_menu.pack(pady=20) # 추천 버튼 recommend_btn = tk.Button( root, text="🎶 음악 추천받기", command=recommend_music, font=("Helvetica", 14, "bold"), bg="#00cc99", fg="white", padx=20, pady=10, bd=0, cursor="hand2" ) recommend_btn.pack(pady=20) # 결과 출력 result_label = tk.Label( root, text="", font=("Helvetica", 16, "bold"), bg="#121212", fg="white", wraplength=400, justify="center" ) result_label.pack(pady=30) # 하단 문구 footer = tk.Label( root, text="Made with Python 💻", font=("Helvetica", 10), bg="#121212", fg="#666666" ) footer.pack(side="bottom", pady=10) # 실행 root.mainloop()
